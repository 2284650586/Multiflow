91 template <typename T> std::string repr(T const &val) {
92 if constexpr (std::is_same_v<T, bool>) {
93 return val ? "true" : "false";
94 } else if constexpr (std::is_convertible_v<T, std::string_view>) {
95 return '"' + std::string{std::string_view{val}} + '"';
96 } else if constexpr (IsContainer<T>) {
97 std::stringstream out;
98 out << "{";
99 const auto size = val.size();
100 if (size > 1) {
101 out << repr(*val.begin());
102 std::for_each(
103 std::next(val.begin()),
104 std::next(
105 val.begin(),
106 static_cast<typename T::iterator::difference_type>(
107 std::min<std::size_t>(size, repr_max_container_size) - 1)),
108 [&out](const auto &v) { out << " " << repr(v); });
109 if (size <= repr_max_container_size) {
110 out << " ";
111 } else {
112 out << "...";
113 }
114 }
115 if (size > 0) {
116 out << repr(*std::prev(val.end()));
117 }
118 out << "}";
119 return out.str();
120 } else if constexpr (IsStreamable<T>) {
121 std::stringstream out;
122 out << val;
123 return out.str();
124 } else {
125 return "<not representable>";
126 }

205 template <class T, auto Param>
206 inline auto do_from_chars(std::string_view s) -> T {
207 T x;
208 auto [first, last] = pointer_range(s);
209 auto [ptr, ec] = std::from_chars(first, last, x, Param);
210 if (ec == std::errc()) {
211 if (ptr == last) {
212 return x;
213 }
214 throw std::invalid_argument{"pattern does not match to the end"};
215 }
216 if (ec == std::errc::invalid_argument) {
217 throw std::invalid_argument{"pattern not found"};
218 }
219 if (ec == std::errc::result_out_of_range) {
220 throw std::range_error{"not representable"};
221 }
222 return x; // unreachable
223 }

262 template <class T> inline auto do_strtod(std::string const &s) -> T {
263 if (isspace(static_cast<unsigned char>(s[0])) || s[0] == '+') {
264 throw std::invalid_argument{"pattern not found"};
265 }
266
267 auto [first, last] = pointer_range(s);
268 char *ptr;
269
270 errno = 0;
271 auto x = generic_strtod<T>(first, &ptr);
272 if (errno == 0) {
273 if (ptr == last) {
274 return x;
275 }
276 throw std::invalid_argument{"pattern does not match to the end"};
277 }
278 if (errno == ERANGE) {
279 throw std::range_error{"not representable"};
280 }
281 return x; // unreachable
282 }

428 auto action(F &&callable, Args &&... bound_args)
429 -> std::enable_if_t<std::is_invocable_v<F, Args..., std::string const>,
430 Argument &> {

452 template <char Shape, typename T>
453 auto scan() -> std::enable_if_t<std::is_arithmetic_v<T>, Argument &> {
454 static_assert(!(std::is_const_v<T> || std::is_volatile_v<T>),
455 "T should not be cv-qualified");
456 auto is_one_of = [](char c, auto... x) constexpr {
457 return ((c == x) || ...);
458 };
459
460 if constexpr (is_one_of(Shape, 'd') && details::standard_integer<T>) {
461 action(details::parse_number<T, details::radix_10>());
462 } else if constexpr (is_one_of(Shape, 'i') &&
463 details::standard_integer<T>) {
464 action(details::parse_number<T>());
465 } else if constexpr (is_one_of(Shape, 'u') &&
466 details::standard_unsigned_integer<T>) {
467 action(details::parse_number<T, details::radix_10>());
468 } else if constexpr (is_one_of(Shape, 'o') &&
469 details::standard_unsigned_integer<T>) {
470 action(details::parse_number<T, details::radix_8>());
471 } else if constexpr (is_one_of(Shape, 'x', 'X') &&
472 details::standard_unsigned_integer<T>) {
473 action(details::parse_number<T, details::radix_16>());
474 } else if constexpr (is_one_of(Shape, 'a', 'A') &&
475 std::is_floating_point_v<T>) {
476 action(details::parse_number<T, details::chars_format::hex>());
477 } else if constexpr (is_one_of(Shape, 'e', 'E') &&
478 std::is_floating_point_v<T>) {
479 action(details::parse_number<T, details::chars_format::scientific>());
480 } else if constexpr (is_one_of(Shape, 'f', 'F') &&
481 std::is_floating_point_v<T>) {
482 action(details::parse_number<T, details::chars_format::fixed>());
483 } else if constexpr (is_one_of(Shape, 'g', 'G') &&
484 std::is_floating_point_v<T>) {
485 action(details::parse_number<T, details::chars_format::general>());
486 } else {
487 static_assert(alignof(T) == 0, "No scan specification for T");
488 }
489
490 return *this;
491 }

523 template <typename Iterator>
524 Iterator consume(Iterator start, Iterator end,
525 std::string_view used_name = {}) {
526 if (!m_is_repeatable && m_is_used) {
527 throw std::runtime_error("Duplicate argument");
528 }
529 m_is_used = true;
530 m_used_name = used_name;
531
532 const auto num_args_max = m_num_args_range.get_max();
533 const auto num_args_min = m_num_args_range.get_min();
534 std::size_t dist = 0;
535 if (num_args_max == 0) {
536 m_values.emplace_back(m_implicit_value);
537 std::visit([](const auto &f) { f({}); }, m_action);
538 return start;
539 }
540 if ((dist = static_cast<std::size_t>(std::distance(start, end))) >=
541 num_args_min) {
542 if (num_args_max < dist) {
543 end = std::next(start, static_cast<typename Iterator::difference_type>(
544 num_args_max));
545 }
546 if (!m_accepts_optional_like_value) {
547 end = std::find_if(
548 start, end,
549 std::bind(is_optional, std::placeholders::_1, m_prefix_chars));
550 dist = static_cast<std::size_t>(std::distance(start, end));
551 if (dist < num_args_min) {
552 throw std::runtime_error("Too few arguments");
553 }
554 }

556 struct ActionApply {

558 std::transform(first, last, std::back_inserter(self.m_values), f);

560
561 void operator()(void_action &f) {
562 std::for_each(first, last, f);
563 if (!self.m_default_value.has_value()) {
564 if (!self.m_accepts_optional_like_value) {
565 self.m_values.resize(
566 static_cast<std::size_t>(std::distance(first, last)));

568 }
569 }
570
571 Iterator first, last;
572 Argument &self;
573 };
574 std::visit(ActionApply{start, end, *this}, m_action);
575 return end;
576 }
577 if (m_default_value.has_value()) {
578 return start;
579 }
580 throw std::runtime_error("Too few arguments for '" +
581 std::string(m_used_name) + "'.");

630 std::size_t get_arguments_length() const {
631
632 std::size_t names_size = std::accumulate(
633 std::begin(m_names), std::end(m_names), std::size_t(0),
634 [](const auto &sum, const auto &s) { return sum + s.size(); });
635
636 if (is_positional(m_names.front(), m_prefix_chars)) {
637 // A set metavar means this replaces the names
638 if (!m_metavar.empty()) {
639 // Indent and metavar
640 return 2 + m_metavar.size();
641 }
642
643 // Indent and space-separated
644 return 2 + names_size + (m_names.size() - 1);
645 }
646 // Is an option - include both names _and_ metavar
647 // size = text + (", " between names)
648 std::size_t size = names_size + 2 * (m_names.size() - 1);
649 if (!m_metavar.empty() && m_num_args_range == NArgsRange{1, 1}) {
650 size += m_metavar.size() + 1;
651 }
652 return size + 2; // indent
653 }

686 while ((pos = argument.m_help.find('\n', prev)) != std::string::npos) {
687 auto line = help_view.substr(prev, pos - prev + 1);
688 if (first_line) {
689 stream << hspace << line;
690 first_line = false;
691 } else {
692 stream.width(stream_width);
693 stream << name_padding << hspace << line;
694 }
695 prev += pos - prev + 1;
696 }
697 if (first_line) {
698 stream << hspace << argument.m_help;
699 } else {
700 auto leftover = help_view.substr(prev, argument.m_help.size() - prev);
701 if (!leftover.empty()) {
702 stream.width(stream_width);
703 stream << name_padding << hspace << leftover;
704 }
705 }
706
707 // print nargs spec
708 if (!argument.m_help.empty()) {
709 stream << " ";
710 }
711 stream << argument.m_num_args_range;
712
713 if (argument.m_default_value.has_value() &&
714 argument.m_num_args_range != NArgsRange{0, 0}) {
715 stream << "[default: " << argument.m_default_value_repr << "]";
716 } else if (argument.m_is_required) {
717 stream << "[required]";
718 }
719 stream << "\n";
720 return stream;
721 }

829 static auto lookahead(std::string_view s) -> int {
830 if (s.empty()) {
831 return eof;
832 }
833 return static_cast<int>(static_cast<unsigned char>(s[0]));

836 /*
837 * decimal-literal:
838 * '0'
839 * nonzero-digit digit-sequence_opt
840 * integer-part fractional-part
841 * fractional-part
842 * integer-part '.' exponent-part_opt
843 * integer-part exponent-part
844 *
845 * integer-part:
846 * digit-sequence
847 *
848 * fractional-part:
849 * '.' post-decimal-point
850 *
851 * post-decimal-point:
852 * digit-sequence exponent-part_opt
853 *
854 * exponent-part:
855 * 'e' post-e
856 * 'E' post-e
857 *
858 * post-e:
859 * sign_opt digit-sequence
860 *
861 * sign: one of
862 * '+' '-'
863 */
864 static bool is_decimal_literal(std::string_view s) {
865 auto is_digit = [](auto c) constexpr {
866 switch (c) {
867 case '0':
868 case '1':
869 case '2':
870 case '3':
871 case '4':
872 case '5':
873 case '6':
874 case '7':
875 case '8':
876 case '9':
877 return true;
878 default:
879 return false;
880 }
881 };
882
883 // precondition: we have consumed or will consume at least one digit
884 auto consume_digits = [=](std::string_view sd) {
885 // NOLINTNEXTLINE(readability-qualified-auto)
886 auto it = std::find_if_not(std::begin(sd), std::end(sd), is_digit);
887 return sd.substr(static_cast<std::size_t>(it - std::begin(sd)));
888 };
889
890 switch (lookahead(s)) {
891 case '0': {
892 s.remove_prefix(1);
893 if (s.empty()) {
894 return true;
895 }
896 goto integer_part;
897 }
898 case '1':
899 case '2':
900 case '3':
901 case '4':
902 case '5':
903 case '6':
904 case '7':
905 case '8':
906 case '9': {
907 s = consume_digits(s);
908 if (s.empty()) {
909 return true;
910 }
911 goto integer_part_consumed;
912 }
913 case '.': {
914 s.remove_prefix(1);
915 goto post_decimal_point;
916 }
917 default:
918 return false;
919 }
920
921 integer_part:
922 s = consume_digits(s);
923 integer_part_consumed:
924 switch (lookahead(s)) {
925 case '.': {
926 s.remove_prefix(1);
927 if (is_digit(lookahead(s))) {
928 goto post_decimal_point;
929 } else {
930 goto exponent_part_opt;
931 }
932 }
933 case 'e':
934 case 'E': {
935 s.remove_prefix(1);
936 goto post_e;
937 }
938 default:
939 return false;
940 }
941
942 post_decimal_point:
943 if (is_digit(lookahead(s))) {
944 s = consume_digits(s);
945 goto exponent_part_opt;
946 }
947 return false;
948
949 exponent_part_opt:
950 switch (lookahead(s)) {
951 case eof:
952 return true;
953 case 'e':
954 case 'E': {
955 s.remove_prefix(1);
956 goto post_e;
957 }
958 default:
959 return false;
960 }
961
962 post_e:
963 switch (lookahead(s)) {
964 case '-':
965 case '+':
966 s.remove_prefix(1);
967 }
968 if (is_digit(lookahead(s))) {
969 s = consume_digits(s);
970 return s.empty();
971 }
972 return false;
973 }

975 static bool is_optional(std::string_view name,
976 std::string_view prefix_chars) {
977 return !is_positional(name, prefix_chars);
978 }

980 /*
981 * positional:
982 * _empty_
983 * '-'
984 * '-' decimal-literal
985 * !'-' anything
986 */
987 static bool is_positional(std::string_view name,
988 std::string_view prefix_chars) {
989 auto first = lookahead(name);
990
991 if (first == eof) {
992 return true;
993 } else if (prefix_chars.find(static_cast<char>(first)) !=
994 std::string_view::npos) {
995 name.remove_prefix(1);
996 if (name.empty()) {
997 return true;
998 }
999 return is_decimal_literal(name);
1000 }
1001 return true;
1002 }

1475 /*
1476 * Pre-process this argument list. Anything starting with "--", that
1477 * contains an =, where the prefix before the = has an entry in the
1478 * options table, should be split.
1479 */
1480 std::vector<std::string>
1481 preprocess_arguments(const std::vector<std::string> &raw_arguments) const {
1482 std::vector<std::string> arguments{};
1483 for (const auto &arg : raw_arguments) {
1484
1485 const auto argument_starts_with_prefix_chars =
1486 [this](const std::string &a) -> bool {
1487 if (!a.empty()) {
1488
1489 const auto legal_prefix = [this](char c) -> bool {
1490 return m_prefix_chars.find(c) != std::string::npos;
1491 };
1492
1493 // Windows-style
1494 // if '/' is a legal prefix char
1495 // then allow single '/' followed by argument name, followed by an
1496 // assign char, e.g., ':' e.g., 'test.exe /A:Foo'
1497 const auto windows_style = legal_prefix('/');
1498
1499 if (windows_style) {
1500 if (legal_prefix(a[0])) {
1501 return true;
1502 }
1503 } else {
1504 // Slash '/' is not a legal prefix char
1505 // For all other characters, only support long arguments
1506 // i.e., the argument must start with 2 prefix chars, e.g,
1507 // '--foo' e,g, './test --foo=Bar -DARG=yes'
1508 if (a.size() > 1) {
1509 return (legal_prefix(a[0]) && legal_prefix(a[1]));
1510 }
1511 }
1512 }
1513 return false;
1514 };
1515
1516 // Check that:
1517 // - We don't have an argument named exactly this
1518 // - The argument starts with a prefix char, e.g., "--"
1519 // - The argument contains an assign char, e.g., "="
1520 auto assign_char_pos = arg.find_first_of(m_assign_chars);
1521
1522 if (m_argument_map.find(arg) == m_argument_map.end() &&
1523 argument_starts_with_prefix_chars(arg) &&
1524 assign_char_pos != std::string::npos) {
1525 // Get the name of the potential option, and check it exists
1526 std::string opt_name = arg.substr(0, assign_char_pos);
1527 if (m_argument_map.find(opt_name) != m_argument_map.end()) {
1528 // This is the name of an option! Split it into two parts
1529 arguments.push_back(std::move(opt_name));
1530 arguments.push_back(arg.substr(assign_char_pos + 1));
1531 continue;
1532 }
1533 }
1534 // If we've fallen through to here, then it's a standard argument
1535 arguments.push_back(arg);
1536 }
1537 return arguments;
1538 }

1541 * @throws std::runtime_error in case of any invalid argument
1542 */
1543 void parse_args_internal(const std::vector<std::string> &raw_arguments) {
1544 auto arguments = preprocess_arguments(raw_arguments);
1545 if (m_program_name.empty() && !arguments.empty()) {
1546 m_program_name = arguments.front();
1547 }
1548 auto end = std::end(arguments);
1549 auto positional_argument_it = std::begin(m_positional_arguments);
1550 for (auto it = std::next(std::begin(arguments)); it != end;) {
1551 const auto &current_argument = *it;
1552 if (Argument::is_positional(current_argument, m_prefix_chars)) {
1553 if (positional_argument_it == std::end(m_positional_arguments)) {
1554
1555 std::string_view maybe_command = current_argument;
1556
1557 // Check sub-parsers
1558 auto subparser_it = m_subparser_map.find(maybe_command);
1559 if (subparser_it != m_subparser_map.end()) {
1560
1561 // build list of remaining args
1562 const auto unprocessed_arguments =
1563 std::vector<std::string>(it, end);
1564
1565 // invoke subparser
1566 m_is_parsed = true;
1567 m_subparser_used[maybe_command] = true;
1568 return subparser_it->second->get().parse_args(
1569 unprocessed_arguments);
1570 }
1571
1572 throw std::runtime_error(
1573 "Maximum number of positional arguments exceeded");
1574 }
1575 auto argument = positional_argument_it++;
1576 it = argument->consume(it, end);
1577 continue;
1578 }
1579
1580 auto arg_map_it = m_argument_map.find(current_argument);
1581 if (arg_map_it != m_argument_map.end()) {
1582 auto argument = arg_map_it->second;
1583 it = argument->consume(std::next(it), end, arg_map_it->first);
1584 } else if (const auto &compound_arg = current_argument;
1585 compound_arg.size() > 1 &&
1586 is_valid_prefix_char(compound_arg[0]) &&
1587 !is_valid_prefix_char(compound_arg[1])) {
1588 ++it;
1589 for (std::size_t j = 1; j < compound_arg.size(); j++) {
1590 auto hypothetical_arg = std::string{'-', compound_arg[j]};
1591 auto arg_map_it2 = m_argument_map.find(hypothetical_arg);
1592 if (arg_map_it2 != m_argument_map.end()) {
1593 auto argument = arg_map_it2->second;
1594 it = argument->consume(it, end, arg_map_it2->first);
1595 } else {
1596 throw std::runtime_error("Unknown argument: " + current_argument);
1597 }
1598 }
1599 } else {
1600 throw std::runtime_error("Unknown argument: " + current_argument);
1601 }
1602 }
1603 m_is_parsed = true;
1604 }
